1) Introdução

Referência: FLUSP - Introduction to Linux kernel Character Device Drivers

Este tutorial, embora seja o mais curto em termos de ações práticas, é denso em conteúdo conceitual. O foco é entender o que são character devices no Linux, como o kernel os identifica por meio de major e minor numbers, e como escrever um driver simples que expõe um dispositivo de caractere para o userspace.

No Linux, todo dispositivo é representado como um arquivo especial. Os character devices são aqueles que transferem dados byte a byte - como terminais (tty) e dispositivos de memória. Cada driver registra um major number único que o identifica no sistema, e pode gerenciar múltiplas instâncias por meio dos minor numbers.

Para associar IDs de dispositivo às operações de arquivo, o kernel utiliza a estrutura struct cdev, definida em include/linux/cdev.h: 

struct cdev {
	struct kobject kobj;
	struct module *owner;
	const struct file_operations *ops;
	struct list_head list;
	dev_t dev;
	unsigned int count;
} __randomize_layout;

Esta estrutura é o elo entre o identificador do dispositivo (dev_t) e as operações de arquivo (file_operations) que o driver implementa.

2) Inspecionando Dispositivos do Sistema

Listando dispositivos com /proc/devices

O arquivo /proc/devices lista todos os drivers de dispositivo registrados no kernel em execução, agrupados por major number. Executei o comando no host para ver o estado atual: 

cat /proc/devices

Por exemplo, o major 1 corresponde ao driver de memória, enquanto tty e ttyS compartilham o major 4, cada um gerenciando um subconjunto dos minors disponíveis naquele major.

Verificando major e minor com stat

O comando stat permite inspecionar os metadados de qualquer arquivo, incluindo o identificador do dispositivo responsável. Ao executar em uma imagem de disco, o campo Device revela o major e o minor do dispositivo de bloco subjacente:

4) Compilando e Instalando o Módulo

Habilitando via menuconfig

Com o driver integrado ao sistema de build, habilitei-o como módulo via menuconfig: 

make -C "$IIO_TREE" menuconfig

Em seguida, compilei o kernel com o novo módulo:

kw build

Instalando na VM

Com a VM desligada, montei a imagem de disco com guestmount e instalei os módulos compilados diretamente na imagem: 

sudo guestmount --rw --add "${VM_DIR}/arm64_img.qcow2" \
  --mount /dev/vda2 "${VM_DIR}/arm64_rootfs"

sudo --preserve-env make -C "${IIO_TREE}" \
  INSTALL_MOD_PATH="${VM_DIR}/arm64_rootfs" modules_install

Após instalar, desmontei a imagem e iniciei a VM:

sudo guestunmount "${VM_DIR}/arm64_rootfs"
sudo virsh start arm64
kw ssh

Testando a leitura

Para testar a operação de leitura, escrevi um programa em C no host, compilei para ARM64 com o cross-compiler e enviei à VM via SCP: 

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

#define BUF_SIZE 256

/* Example inspired from open(2) */
int main(int argc, const char **argv)
{
	char buf[BUF_SIZE];
	int fd;

	if (argc < 2)
		return -22;

	fd = open(argv[1], O_RDONLY);
	read(fd, buf, BUF_SIZE);
	printf("Read buffer: %s\n", buf);
	close(fd);
}
(LK-DEV) $ aarch64-linux-gnu-gcc read_prog.c -o read_prog
(LK-DEV) $ scp read_prog root@192.168.122.56:~/

Testando a escrita

De forma análoga, criei um write_prog.c para testar a operação de escrita no dispositivo: 

#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <errno.h>

#define BUF_SIZE 256

int main(int argc, const char **argv)
{
	char buf[BUF_SIZE];
	int errsv;
	int ret;
	int fd;

	if (argc < 2)
		return -EINVAL;

	sprintf(buf, "A new message for simple_char.");
	fd = open(argv[1], O_RDWR);
	ret = write(fd, buf, BUF_SIZE);
	if (ret < 0) {
		errsv = errno;
		printf("Error: %d", errsv);
	}
	printf("wrote %d bytes to buffer\n", ret);
	return close(fd);
}
(LK-DEV) $ aarch64-linux-gnu-gcc write_prog.c -o write_prog
(LK-DEV) $ scp write_prog root@192.168.122.56:~/

Executando dentro da VM:

root@localhost:~# ./write_prog simple_char_node
wrote 256 bytes to buffer

As operações de leitura e escrita funcionaram corretamente por meio do arquivo de dispositivo.

6) Exercícios

Exercício 1 - Exibir major e minor ao inicializar o módulo

O primeiro exercício propõe adicionar ao simple_char_init() um log que exiba o par major:minor alocados pelo driver. Um detalhe importante: o dev_id só é preenchido após a chamada a alloc_chrdev_region(), portanto as macros MAJOR() e MINOR() só produzem valores corretos depois dessa etapa.

Após compilar, instalar e carregar o módulo na VM, o resultado foi:

root@localhost:~# modprobe simple_char
root@localhost:~# dmesg | tail
[   71.017271] Initialize simple_char module.
[   71.017886] 238:0 # (--> Major:Minor)

Exercício 2 - Device private data com múltiplos minors

O segundo exercício é mais elaborado: estender o driver para suportar múltiplos minor numbers, cada um com seu próprio buffer de dados independente. A proposta é alocar os buffers sob demanda no open e liberá-los no exit, identificando o minor correto a partir do inode do arquivo.

Primeiramente, alterei a constante de minors de 1 para 2 e declarei um array de buffers: 

#define MINOR_NUMS 2 /* 1 -> 2 */

/* One buffer per minor */
static char *s_bufs[MINOR_NUMS];

Para extrair o minor dentro de cada operação, o exercício sugere usar inode->i_rdev no open e release, e file->f_inode->i_rdev nas funções read e write. Durante a implementação, tentei usar strcpy com formatação, o que causou um erro de compilação: 

drivers/char/simple_char.c:34:17: error: too many arguments to function 'strcpy'

Pesquisando, encontrei snprintf() como a alternativa correta para formatar strings no kernel. Com isso, as quatro operações de arquivo ficaram assim: 

static int simple_char_open(struct inode *inode, struct file *file)
{
	dev_t dev = inode->i_rdev;
	unsigned int minor = MINOR(dev);

	pr_info("%s: %s minor %u\n", KBUILD_MODNAME, __func__, minor);

	if (!s_bufs[minor]) {
		s_bufs[minor] = kmalloc(S_BUFF_SIZE, GFP_KERNEL);
		if (!s_bufs[minor])
			return -ENOMEM;

		snprintf(s_bufs[minor], S_BUFF_SIZE,
			 "This is data from simple_char minor %u buffer.\n", minor);
		pr_info("%s: allocated buffer for minor %u\n", KBUILD_MODNAME, minor);
	}
	return 0;
}

static ssize_t simple_char_read(struct file *file, char __user *buffer,
				size_t count, loff_t *ppos)
{
	int n_bytes;
	unsigned int minor = MINOR(file->f_inode->i_rdev);

	pr_info("%s: %s minor %u about to read %ld bytes from position %lld\n",
		KBUILD_MODNAME, __func__, minor, count, *ppos);
	n_bytes = count - copy_to_user(buffer, s_bufs[minor] + *ppos, count);
	*ppos += n_bytes;
	return n_bytes;
}

static ssize_t simple_char_write(struct file *file, const char __user *buffer,
				size_t count, loff_t *ppos)
{
	int n_bytes;
	unsigned int minor = MINOR(file->f_inode->i_rdev);

	pr_info("%s: %s minor %u about to write %ld bytes to position %lld\n",
		KBUILD_MODNAME, __func__, minor, count, *ppos);
	n_bytes = count - copy_from_user(s_bufs[minor] + *ppos, buffer, count);
	return n_bytes;
}

static int simple_char_release(struct inode *inode, struct file *file)
{
	pr_info("%s: %s minor %u\n", KBUILD_MODNAME, __func__, MINOR(inode->i_rdev));
	return 0;
}

As funções de init e exit foram atualizadas para registrar os dois minors e liberar corretamente os buffers alocados: 

static int __init simple_char_init(void)
{
	int ret;

	pr_info("Initialize %s module.\n", KBUILD_MODNAME);

	ret = alloc_chrdev_region(&dev_id, 0, MINOR_NUMS, "simple_char");
	if (ret < 0)
		return ret;

	pr_info("%s: major %u, minors 0-%d\n", KBUILD_MODNAME, MAJOR(dev_id), MINOR_NUMS - 1);

	s_cdev = cdev_alloc();
	s_cdev->ops = &simple_char_fops;
	s_cdev->owner = simple_char_fops.owner;

	return cdev_add(s_cdev, dev_id, MINOR_NUMS);
}

static void __exit simple_char_exit(void)
{
	int i;
	for (i = 0; i < MINOR_NUMS; i++) {
		if (s_bufs[i]) {
			kfree(s_bufs[i]);
			pr_info("%s: buffer minor %d free\n", KBUILD_MODNAME, i);
		}
	}

	cdev_del(s_cdev);
	unregister_chrdev_region(dev_id, MINOR_NUMS);

	pr_info("%s exiting.\n", KBUILD_MODNAME);
}

Após compilar e instalar o módulo com o mesmo fluxo dos tutoriais anteriores, criei dois node devices na VM (um por minor) e testei operações de leitura e escrita:

root@localhost:~# dmesg | tail

[   76.220987] simple_char: major 238, minors 0-1

root@localhost:~# mknod simple_char_node0 c 238 0
root@localhost:~# mknod simple_char_node1 c 238 1

root@localhost:~# echo "hello from minor 1" > simple_char_node1
root@localhost:~# head -1 simple_char_node1
